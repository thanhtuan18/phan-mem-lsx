async function handleLogin() { const u = document.getElementById('user').value.trim(); const p = document.getElementById('pass').value.trim(); const btn = document.getElementById('login-btn'); const err = document.getElementById('error-box'); console.log("--- BẮT ĐẦU DEBUG LOGIN ---"); btn.innerText = "Đang kiểm tra kết nối..."; btn.disabled = true; try { // KIỂM TRA 1: Thử ping đến domain để xem server có sống không console.log("1. Kiểm tra trạng thái Server..."); const ping = await fetch('https://tiendosanxuat.technoup.info/').catch(e => { throw new Error("KHÔNG THỂ CHẠM ĐẾN DOMAIN. Có thể do: 1. SSL chưa bật, 2. Firewall hosting chặn, 3. Sai URL."); }); // KIỂM TRA 2: Gửi request đăng nhập thực tế console.log("2. Đang gửi yêu cầu xác thực..."); const response = await fetch('https://tiendosanxuat.technoup.info/api/auth/login', { method: 'POST', headers: { 'Content-Type': 'application/json' }, body: JSON.stringify({ username: u, password: p }) }); console.log("Mã trạng thái HTTP:", response.status); if (!response.ok) { const errorText = await response.text(); console.error("Lỗi từ Server trả về:", errorText); alert(`Lỗi Server (${response.status}): ${errorText.substring(0, 50)}`); return; } const result = await response.json(); console.log("Kết quả nhận được:", result); if (result.status === "success") { alert("Kết nối thành công!"); localStorage.setItem('user_logged', JSON.stringify({...result.user, token: result.token})); location.reload(); } else { alert("Server từ chối: " + result.message); } } catch (e) { console.error("--- CHI TIẾT LỖI DEBUG ---"); console.error("Message:", e.message); // Kiểm tra xem có phải lỗi CORS không if (e.message.includes('fetch')) { alert("LỖI KẾT NỐI: Có thể do CORS hoặc SSL. Hãy nhấn F12, xem tab Console để thấy lỗi đỏ."); } else { alert("LỖI: " + e.message); } } finally { btn.innerText = "ĐĂNG NHẬP"; btn.disabled = false; } }
