async function handleLogin() { const u = document.getElementById('user').value.trim(); const p = document.getElementById('pass').value.trim(); const btn = document.getElementById('login-btn'); const errBox = document.getElementById('error-box'); // Tạo khu vực hiển thị log debug ngay trên giao diện if (!document.getElementById('debug-log')) { const logDiv = document.createElement('div'); logDiv.id = 'debug-log'; logDiv.style = "background:#000; color:#0f0; font-family:monospace; padding:10px; font-size:11px; margin-top:10px; max-height:150px; overflow:auto; text-align:left;"; document.querySelector('.login-card').appendChild(logDiv); } const logger = document.getElementById('debug-log'); const addLog = (msg) => logger.innerHTML += `> ${msg}
`; logger.innerHTML = ""; // Reset log addLog("Bắt đầu tiến trình Debug..."); try { // BƯỚC 1: Kiểm tra kết nối cơ bản (Pre-flight) addLog("Bước 1: Kiểm tra DNS & SSL..."); await fetch('https://tiendosanxuat.technoup.info/', { mode: 'no-cors' }); addLog("✓ Đã chạm được tới Domain."); // BƯỚC 2: Gửi request chính thức addLog("Bước 2: Gửi yêu cầu POST tới API..."); const response = await fetch('https://tiendosanxuat.technoup.info/api/auth/login', { method: 'POST', headers: { 'Content-Type': 'application/json' }, body: JSON.stringify({ username: u, password: p }) }); addLog(`Bước 3: Nhận phản hồi. LỖI NGHIÊM TRỌNG !!!"); addLog(`Chi tiết: ${e.message}`); if (e.message.includes('fetch')) { addLog("Gợi ý: Lỗi này 90% do CORS ở Backend chưa cho phép GitHub Pages."); } } }
